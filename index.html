function doPost(e) { try { const body = parsePostBody_(e); // HTML側から送られてくる 'adjustMiles' と、本来の 'miles' 両方に対応できるように修正 let action = body.action; if (action === "adjustMiles") { action = "miles"; // アクション名をスプシ用に変換 } const userId = body.id || body.userId; // どちらの記述でもIDを認識できるように調整 const amount = Number(body.amount || 0); const adminPin = String(body.adminPin || body.masterPin || "").trim(); // masterPinにも対応 if (adminPin !== ADMIN_MASTER_PIN) { return output_(fail_("管理者認証エラー：PINが違います")); } const sheet = getSheet_(SHEETS.users); const user = findUserById_(userId); if (!user) return output_(fail_("ユーザーが見つかりません")); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(String); const col = getUserColumns_(headers); let targetColumnIndex = -1; let currentVal = 0; let actionName = ""; // HTML側のtype（'payment'か'charge'か）を判別して金額の正負を自動調整 const kind = body.type || ""; let finalAmount = amount; if (kind === "payment" && amount > 0) { finalAmount = -amount; // マイルを引く場合はマイナスにする } if (action === "miles") { targetColumnIndex = col.miles + 1; currentVal = user.miles; actionName = finalAmount >= 0 ? "チャージ" : "決済"; } else if (action === "tickets") { targetColumnIndex = col.tickets + 1; currentVal = user.tickets; actionName = finalAmount >= 0 ? "チケット付与" : "チケット消費"; } else { return output_(fail_("無効なポストデータです: " + action)); } const newVal = currentVal + finalAmount; if (newVal < 0) return output_(fail_("残高が不足しています")); // スプレッドシートの更新 sheet.getRange(user.rowNumber, targetColumnIndex).setValue(newVal); // 履歴の追記 appendHistory_(userId, actionName, Math.abs(finalAmount)); // 最新のユーザー情報を返す const updatedUser = findUserById_(userId); return output_(ok_({ user: publicUser_(updatedUser) })); } catch (err) { return output_(fail_("システムエラー: " + err.message)); } }
